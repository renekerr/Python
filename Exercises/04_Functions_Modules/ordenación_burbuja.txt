//Procedimiento para intercambiar valores
SubProceso IntercValores(_OrdArregloNumeros Por Referencia, _i, _min)
	Definir Temporal como Numerico;
	
	Temporal <- _OrdArregloNumeros[_i];
	_OrdArregloNumeros[_i] <- _OrdArregloNumeros[_min];
	_OrdArregloNumeros[_min] <- Temporal;
	
	Para c <- 0 hasta 4 Hacer
		Escribir Sin Saltar _OrdArregloNumeros[c], ' ';
	FinPara
	Escribir ''; //Salto de línea	
	
	
FinSubProceso



//Procedimiento para ordenar valores
//Ordenaión por Burbuja.
//Se basa en ir comparando pares de números e ir ordenándolos, de esta manera tendremos a la izq. el menor y a la der. el mayor de estos dos números
SubProceso OrdenarArreglo(OrdArregloNumeros Por Referencia, _nElem)
	
	Definir i como Numerico; 
	Definir intercambiados como Logico;

	Repetir
		//Se define la var intercambiados como FALSO
		//Se recorre todo el arreglo en busca de posibles intercambios, de no ser el caso la var intercambio continuará siendo FALSO y se acabaría el repetir.
		intercambiados <- Falso;
		
		//Recorrer los valore del arreglo
		Para i <- 1 hasta _nElem - 1 hacer
			
			// Si el valor del arreglo en la posición i es menor que el arreglo en la posición i-1, entonces hacemos es intercambio de valores
			Si OrdArregloNumeros[i-1] > OrdArregloNumeros[i] Entonces
				IntercValores(OrdArregloNumeros, i-1, i);
				intercambiados <- Verdadero; // Cambiamos el valor a Verdadero porque hemos hecho un cambio
			FinSi
			
		FinPara
		
	Hasta Que (intercambiados = Falso)
	
FinSubProceso



//Procedimiento para mostrar valores recién generados
SubProceso MostrarArreglo(_ArregloNumeros Por Referencia, _nElem) 
	Definir indice como Entero;
	
	Para indice<-0 Hasta _nElem-1 Hacer
		Escribir Sin Saltar _ArregloNumeros[indice], ' ';
	FinPara
	Escribir ''; //Salto de línea	
FinSubProceso 




Proceso ArreglosOrdenacionPorBurbuja //Proceso Principal
	Definir ArregloNumeros, nElem como Entero; //Definición de un arreglo [n]
	nElem <- 5;
	
	Dimension ArregloNumeros[nElem];
	
	
	Para i<-0 Hasta nElem-1 Hacer
		
		ArregloNumeros[i] <- azar(50); //Genera números aleatorios entre 0-49
		
	Fin Para
	
	
	MostrarArreglo(ArregloNumeros, nElem);
	OrdenarArreglo(ArregloNumeros, nElem);
	
FinProceso
